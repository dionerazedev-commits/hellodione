// Student Class Implementation
class Student {
private:
    int id;
    string name;
    vector grades;
    
public:
    Student(int i, string n) : id(i), name(n) {}
    
    void addGrade(double g) {
        grades.push_back(g);
    }
    
    double calculateGPA() {
        if (grades.empty()) return 0.0;
        double sum = 0;
        for (double g : grades) sum += g;
        return sum / grades.size();
    }
    
    void display() {
        cout << "ID: " << id << " | Name: " << name 
             << " | GPA: " << calculateGPA() << endl;
    }
};

Pseudocode and Flowcharts

Proper planning and documentation are crucial for successful software development. Here's my approach to algorithm design:

ALGORITHM: Binary Search Implementation

INPUT: sorted_array[], target_value, array_size
OUTPUT: index of target_value or -1 if not found

BEGIN
    SET left = 0
    SET right = array_size - 1
    
    WHILE left <= right DO
        SET mid = left + (right - left) / 2
        
        IF sorted_array[mid] == target_value THEN
            RETURN mid
        ELSE IF sorted_array[mid] < target_value THEN
            SET left = mid + 1
        ELSE
            SET right = mid - 1
        END IF
    END WHILE
    
    RETURN -1
END
